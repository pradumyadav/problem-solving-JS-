Q1- Find the Smaller Angle?

var Minimal_Angle = (h, m) =>

{

h=h%12;

let a=(h*360)/12+(m*360)/(12*60);

let b=(m*360)/(60);

let ans=Math.abs(a-b);

if(ans>180){

ans=360-ans;

}

return ans;

};

Q2- Check whether the year is Leap year or not.

var Check_Leap = (year) =>

{

if((year%4===0) && (year%100!==0)){

return 'Leap Year';

}

else if(year%400===0){

return 'Leap Year';

}

else{

return 'Non Leap Year';

}

};

Q3- Perfect Number Check.

var Perfect_Check = (N) =>

{

if(N===1){

return 'YES';

}

let sum=0;

for(let i=0;i<=(N/2);i++){

if(N%i===0){

sum=sum+i;

}

}

if(sum!==0 && N===sum){

return 'YES';

}

else{

return 'NO';

}

};
